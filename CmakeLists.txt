cmake_minimum_required(VERSION 3.31)

project(OpenGl)

set(CMAKE_BUILD_TYPE Release)



set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)

file(GLOB SOURCES "src/*.cpp")

add_executable(out ${SOURCES})

target_include_directories(out PUBLIC ${CMAKE_SOURCE_DIR}/include)
target_include_directories(out PUBLIC ${CMAKE_SOURCE_DIR}/include/glm-0.9.9.7/glm)
target_include_directories(out PUBLIC ${CMAKE_SOURCE_DIR}/include/stb)



if (CMAKE_SYSTEM_NAME STREQUAL "Windows")
    target_link_libraries(out "${CMAKE_SOURCE_DIR}/statLib/libglfw3WIN.a")
    target_link_libraries(out "${CMAKE_SOURCE_DIR}/statLib/libgladWIN.a")
else ()
    target_link_libraries(out
    "-framework CoreFoundation"
    "-framework IOKit"
    "-framework Cocoa"
)
    target_link_libraries(out "${CMAKE_SOURCE_DIR}/statLib/libglfw3MAC.a")
    target_link_libraries(out "${CMAKE_SOURCE_DIR}/statLib/libgladMAC.a")
endif()
